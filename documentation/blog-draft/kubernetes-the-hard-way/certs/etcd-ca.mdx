<summary>ETCD CA</summary>

```bash
mkdir -p ${BASE_K8S_PATH}/pki/etcd

export ETCD_CA_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/ca.key"
export ETCD_CA_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/ca.crt"
export ETCD_CA_CSR_PATH="${BASE_K8S_PATH}/openssl/etcd-ca.conf"
export ETCD_CA_CRT_CN="etcd-ca"
export ETCD_CA_KEY_SIZE=2048
```

<Tabs 
    defaultValue='openssl'
    values={[
        { label: "openssl", value: "openssl" },
    ]}>
    <TabItem value='openssl'>

      ```bash
      mkdir -p ${BASE_K8S_PATH}/openssl
      ```

      ```bash
      cat <<EOF > ${ETCD_CA_CSR_PATH}
      [req]
      distinguished_name = req_distinguished_name
      x509_extensions    = v3_ca
      prompt             = no

      [req_distinguished_name]
      CN = "${ETCD_CA_CRT_CN}"

      [v3_ca]
      keyUsage = critical, keyCertSign, keyEncipherment, digitalSignature
      basicConstraints = critical,CA:TRUE
      EOF
      ```

      ```bash
      openssl genrsa \
        -out "${ETCD_CA_KEY_PATH}" ${ETCD_CA_KEY_SIZE}

      openssl req \
        -x509 \
        -new \
        -nodes \
        -key "${ETCD_CA_KEY_PATH}" \
        -sha256 \
        -days 3650 \
        -out "${ETCD_CA_CRT_PATH}" \
        -config "${ETCD_CA_CSR_PATH}"
      ```

    </TabItem>
