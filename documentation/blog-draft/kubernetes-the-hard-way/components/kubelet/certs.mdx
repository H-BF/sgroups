### Переменные окружения сертификатов

```bash
export CERTIFICATE_KEY_SIZE=2048
export KUBELET_CURRENT_CLIENT_CRT_PATH="${BASE_KUBELET_PATH}/pki/kubelet-client-current.pem"
export KUBELET_CURRENT_SERVER_CRT_PATH="${BASE_KUBELET_PATH}/pki/kubelet-server-current.pem"

export KUBELET_CLIENT_KEY_PATH="${BASE_KUBELET_PATH}/pki/kubelet-client-key.pem"
export KUBELET_CLIENT_CRT_PATH="${BASE_KUBELET_PATH}/pki/kubelet-client.pem"
export KUBELET_CLIENT_CRT_CONF="${BASE_K8S_PATH}/openssl/kubelet-client.conf"
export KUBELET_CLIENT_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/kubelet-client.csr"

export KUBELET_SERVER_KEY_PATH="${BASE_KUBELET_PATH}/pki/kubelet-server-key.pem"
export KUBELET_SERVER_CRT_PATH="${BASE_KUBELET_PATH}/pki/kubelet-server.pem"
export KUBELET_SERVER_CRT_CONF="${BASE_K8S_PATH}/openssl/kubelet-server.conf"
export KUBELET_SERVER_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/kubelet-server.csr"

```

### Инструкция создания сертификатов

    <TabItem value='openssl'>

      #### Конфигурация `openssl` kubelet-client

      ```bash
      cat <<EOF > ${KUBELET_CLIENT_CRT_CONF}
      [ req ]
      default_bits        = ${CERTIFICATE_KEY_SIZE}
      prompt              = no
      default_md          = sha256
      distinguished_name  = dn

      [ dn ]
      CN = system:node:${FULL_HOST_NAME}
      O  = system:nodes

      [ v3_ext ]
      authorityKeyIdentifier=keyid,issuer:always
      basicConstraints=CA:FALSE
      keyUsage=keyEncipherment,dataEncipherment
      extendedKeyUsage=clientAuth
      EOF
      ```

      #### Конфигурация `openssl` kubelet-server

      ```bash
      cat <<EOF > ${KUBELET_SERVER_CRT_CONF}
      [ req ]
      default_bits        = ${CERTIFICATE_KEY_SIZE}
      prompt              = no
      default_md          = sha256
      distinguished_name  = dn
      req_extensions      = req_ext

      [ req_ext ]
      subjectAltName = @alt_names

      [ alt_names ]
      DNS.1 = localhost
      DNS.2 = ${HOST_NAME}
      DNS.3 = ${FULL_HOST_NAME}
      IP.1  = 127.0.0.1
      IP.2  = 0:0:0:0:0:0:0:1
      IP.3  = ${LOCAL_ADDRESS}

      [ dn ]
      CN = "system:node:${FULL_HOST_NAME}"
      O  = "system:nodes"

      [ v3_ext ]
      authorityKeyIdentifier=keyid,issuer:always
      basicConstraints=CA:FALSE
      keyUsage=keyEncipherment,dataEncipherment
      extendedKeyUsage=serverAuth
      subjectAltName=@alt_names
      EOF
      ```

      #### Генерация приватных ключей сертификатов

      ```bash
      openssl genrsa \
        -out "${KUBELET_CLIENT_KEY_PATH}" ${CERTIFICATE_KEY_SIZE}
      openssl genrsa \
        -out "${KUBELET_SERVER_KEY_PATH}" ${CERTIFICATE_KEY_SIZE}
      ```

      #### Генерация CSR будущих сертификатов

      ```bash
      openssl req \
        -new \
        -key "${KUBELET_CLIENT_KEY_PATH}" \
        -out "${KUBELET_CLIENT_CSR_PATH}" \
        -config "${KUBELET_CLIENT_CRT_CONF}"

      openssl req \
        -new \
        -key "${KUBELET_SERVER_KEY_PATH}" \
        -out "${KUBELET_SERVER_CSR_PATH}" \
        -config "${KUBELET_SERVER_CRT_CONF}"
      ```

      #### Подпись CSR ключом удостоверяющего центра (CA)

      ```bash
      openssl x509 \
        -req \
        -days 365 \
        -sha256 \
        -outform PEM \
        -CA "${KUBERNETES_CA_CRT_PATH}" \
        -CAkey "${KUBERNETES_CA_KEY_PATH}" \
        -CAcreateserial \
        -in "${KUBELET_CLIENT_CSR_PATH}" \
        -out "${KUBELET_CLIENT_CRT_PATH}" \
        -extensions v3_ext \
        -extfile "${KUBELET_CLIENT_CRT_CONF}"

      openssl x509 \
        -req \
        -days 365 \
        -sha256 \
        -outform PEM \
        -CA "${KUBERNETES_CA_CRT_PATH}" \
        -CAkey "${KUBERNETES_CA_KEY_PATH}" \
        -CAcreateserial \
        -in "${KUBELET_SERVER_CSR_PATH}" \
        -out "${KUBELET_SERVER_CRT_PATH}" \
        -extensions v3_ext \
        -extfile "${KUBELET_SERVER_CRT_CONF}"

      ```

      ```bash
      cat "${KUBELET_CLIENT_CRT_PATH}" "${KUBELET_CLIENT_KEY_PATH}" >> "${KUBELET_CURRENT_CLIENT_CRT_PATH}"
      cat "${KUBELET_SERVER_CRT_PATH}" "${KUBELET_SERVER_KEY_PATH}" >> "${KUBELET_CURRENT_SERVER_CRT_PATH}"
      ```

### Инструкция создания Kubeconfig
```bash

kubectl config set-cluster kubernetes \
  --certificate-authority="${KUBERNETES_CA_CRT_PATH}" \
  --embed-certs=true \
  --server=https://127.0.0.1:6443 \
  --kubeconfig=${BASE_K8S_PATH}/kubelet.conf

kubectl config set-credentials system:node:${FULL_HOST_NAME} \
  --client-certificate=${KUBELET_CLIENT_CRT_PATH} \
  --client-key=${KUBELET_CLIENT_KEY_PATH} \
  --embed-certs=true \
  --kubeconfig=${BASE_K8S_PATH}/kubelet.conf

kubectl config set-context default \
  --cluster=kubernetes \
  --user=system:node:${FULL_HOST_NAME} \
  --kubeconfig=${BASE_K8S_PATH}/kubelet.conf

kubectl config use-context default --kubeconfig=${BASE_K8S_PATH}/kubelet.conf
```
