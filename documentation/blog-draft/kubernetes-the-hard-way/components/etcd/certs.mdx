<summary>ETCD Certificates</summary>

```bash
mkdir -p ${BASE_K8S_PATH}/pki
mkdir -p ${BASE_K8S_PATH}/openssl/csr

export CERTIFICATE_KEY_SIZE=2048

export ETCD_CLIENT_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/healthcheck-client.key"
export ETCD_CLIENT_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/healthcheck-client.crt"
export ETCD_CLIENT_CRT_CONF="${BASE_K8S_PATH}/openssl/healthcheck-client.conf"
export ETCD_CLIENT_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/etcd-client.csr"

export ETCD_SERVER_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/server.key"
export ETCD_SERVER_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/server.crt"
export ETCD_SERVER_CRT_CONF="${BASE_K8S_PATH}/openssl/etcd-server.conf"
export ETCD_SERVER_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/etcd-server.csr"

export ETCD_PEER_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/peer.key"
export ETCD_PEER_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/peer.crt"
export ETCD_PEER_CRT_CONF="${BASE_K8S_PATH}/openssl/etcd-server.conf"
export ETCD_PEER_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/etcd-peer.csr"
```

      #### Описываем инструкции:

      ```bash
      cat <<EOF > ${ETCD_CLIENT_CRT_CONF}
      [ req ]
      default_bits        = 2048
      prompt              = no
      default_md          = sha256
      distinguished_name  = dn

      [ dn ]
      CN = kube-etcd-healthcheck-client

      [ v3_ext ]
      authorityKeyIdentifier=keyid,issuer:always
      basicConstraints=CA:FALSE
      keyUsage=keyEncipherment,dataEncipherment
      extendedKeyUsage=clientAuth
      EOF
      ```

      ```bash
      cat <<EOF > ${ETCD_SERVER_CRT_CONF}
      [ req ]
      default_bits        = 2048
      prompt              = no
      default_md          = sha256
      distinguished_name  = dn
      req_extensions      = req_ext

      [ req_ext ]
      subjectAltName = @alt_names

      [ alt_names ]
      DNS.1 = localhost
      DNS.2 = ${HOST_NAME}
      DNS.3 = ${FULL_HOST_NAME}
      IP.1  = 127.0.0.1
      IP.2  = 0:0:0:0:0:0:0:1
      IP.3  = ${LOCAL_ADDRESS}

      [ dn ]
      CN = "${FULL_HOST_NAME}"

      [ v3_ext ]
      authorityKeyIdentifier=keyid,issuer:always
      basicConstraints=CA:FALSE
      keyUsage=keyEncipherment,dataEncipherment
      extendedKeyUsage=serverAuth,clientAuth
      subjectAltName=@alt_names
      EOF
      ```

      ```bash
      cat <<EOF > ${ETCD_PEER_CRT_CONF}
      [ req ]
      default_bits        = 2048
      prompt              = no
      default_md          = sha256
      distinguished_name  = dn
      req_extensions      = req_ext

      [ req_ext ]
      subjectAltName = @alt_names

      DNS.1 = localhost
      DNS.2 = ${HOST_NAME}

      IP.2  = 0:0:0:0:0:0:0:1
      IP.3  = ${LOCAL_ADDRESS}

      [ dn ]
      CN = "${FULL_HOST_NAME}"

      basicConstraints=CA:FALSE
      keyUsage=keyEncipherment,dataEncipherment
      extendedKeyUsage=serverAuth,clientAuth

      ```

      #### Создаем приватные ключи:

      ```bash
      openssl genrsa \
        -out "${ETCD_CLIENT_KEY_PATH}" ${CERTIFICATE_KEY_SIZE}
      openssl genrsa \
        -out "${ETCD_SERVER_KEY_PATH}" ${CERTIFICATE_KEY_SIZE}
      openssl genrsa \
        -out "${ETCD_PEER_KEY_PATH}" ${CERTIFICATE_KEY_SIZE}
      ```

      #### Создаем CSR:

      ```bash
      openssl req \
        -new \
        -key "${ETCD_CLIENT_KEY_PATH}" \
        -out "${ETCD_CLIENT_CSR_PATH}" \
        -config "${ETCD_CLIENT_CRT_CONF}"

      openssl req \
        -new \
        -key "${ETCD_SERVER_KEY_PATH}" \
        -out "${ETCD_SERVER_CSR_PATH}" \
        -config "${ETCD_SERVER_CRT_CONF}"

      openssl req \
        -new \
        -key "${ETCD_PEER_KEY_PATH}" \
        -out "${ETCD_PEER_CSR_PATH}" \
        -config "${ETCD_PEER_CRT_CONF}"
      ```

      #### Подписываем CSR:

      ```bash
      openssl x509 \
        -req \
        -days 365 \
        -sha256 \
        -CA "${ETCD_CA_CRT_PATH}" \
        -CAkey "${ETCD_CA_KEY_PATH}" \
        -CAcreateserial \
        -in "${ETCD_CLIENT_CSR_PATH}" \
        -out "${ETCD_CLIENT_CRT_PATH}" \
        -extensions v3_ext \
        -extfile "${ETCD_CLIENT_CRT_CONF}"

      openssl x509 \
        -req \
        -days 365 \
        -sha256 \
        -CA "${ETCD_CA_CRT_PATH}" \
        -CAkey "${ETCD_CA_KEY_PATH}" \
        -CAcreateserial \
        -in "${ETCD_SERVER_CSR_PATH}" \
        -out "${ETCD_SERVER_CRT_PATH}" \
        -extensions v3_ext \
        -extfile "${ETCD_SERVER_CRT_CONF}"

      openssl x509 \
        -req \
        -days 365 \
        -sha256 \
        -CA "${ETCD_CA_CRT_PATH}" \
        -CAkey "${ETCD_CA_KEY_PATH}" \
        -CAcreateserial \
        -in "${ETCD_PEER_CSR_PATH}" \
        -out "${ETCD_PEER_CRT_PATH}" \
        -extensions v3_ext \
        -extfile "${ETCD_PEER_CRT_CONF}"
      ```
