#### Пример использования

```hcl
resource "sgroups_groups" "groups" {
    items = {
        key = {
            name           = "sg-example"
            networks       = ["10.0.0.0/24", "11.0.0.0/24"]
            logs           = true
            trace          = true
            default_action = "ACCEPT"
            icmp = {
                logs = true
                trace = true
                type = [0, 8]
                action = "RULE_ACCEPT"
                priority = "0"
            }
            icmp6 = {
                logs = true
                trace = true
                type = [0, 8]
                action = "RULE_ACCEPT"
                priority = "0"
            }
        }
    }
}
```
